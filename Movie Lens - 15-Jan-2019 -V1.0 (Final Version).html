Handling the Release column to extract features as Release Decade¶

In [43]:
# lets find out Range of release years of Movie Data
print(MasterDF.Release_year.min())
print(MasterDF.Release_year.max())

#print(bins)
#============================================================================
# 1930s : Movie release year <= 1930
# 1940s : Movie release year <= 1940 and > 1930
# 1950s : Movie release year <= 1950 and > 1940
# 1960s : Movie release year <= 1960 and > 1950
# 1970s : Movie release year <= 1970 and > 1960
# 1980s : Movie release year <= 1980 and > 1970
# 1990s : Movie release year <= 1990 and > 1980
# 2000s : Movie release year <= 2000 and > 1990
#============================================================================
cut_bins = [1915,1930,1940,1950,1960,1970,1980,1990,2000]
cut_labels = ['1930s','1940s','1950s','1960s','1970s','1980s','1990s','2000s']
MasterDF["Release_Decade"] = pd.cut(MasterDF.Release_year,bins=cut_bins,labels=cut_labels)

# Now Replace using numeric Values
MasterDF.Release_Decade.replace(['1930s','1940s','1950s','1960s','1970s','1980s','1990s','2000s'],
                                                          [1,2,3,4,5,6,7,8],inplace=True)
